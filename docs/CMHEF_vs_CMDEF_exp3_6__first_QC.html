# Create a summary table of singlets and doublets
doublet_table <- do.call(
  rbind,
  lapply(
    names(sample_list),
    function(nm) {
      
      tab <- table(
        sample_list[[nm]]$scDblFinder.class
      )
      
      singlet_count <- if (
        "singlet" %in% names(tab)
      ) {
        as.numeric(tab["singlet"])
      } else {
        0
      }
      
      doublet_count <- if (
        "doublet" %in% names(tab)
      ) {
        as.numeric(tab["doublet"])
      } else {
        0
      }
      
      total_count <- sum(tab)
      
      data.frame(
        Sample = nm,
        Singlet = singlet_count,
        Doublet = doublet_count,
        Total = total_count,
        Doublet_percent = round(
          100 * doublet_count / total_count,
          1
        )
      )
    }
  )
)
